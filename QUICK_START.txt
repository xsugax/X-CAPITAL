🚀 DEPLOYMENT STEPS IN ORDER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 1: SSH to your server
        ssh root@your_server_ip
        cd /tmp

STEP 2: Clone X-CAPITAL
        git clone https://github.com/xsugax/X-CAPITAL.git
        cd X-CAPITAL

STEP 3: Get SSL certificate (1 minute)
        sudo certbot certonly --standalone -d xcapital.investments
        sudo cp /etc/letsencrypt/live/xcapital.investments/fullchain.pem ssl/xcapital.investments.crt
        sudo cp /etc/letsencrypt/live/xcapital.investments/privkey.pem ssl/private/xcapital.investments.key
        sudo chown $USER:$USER ssl/xcapital.investments.crt ssl/private/xcapital.investments.key

STEP 4: Add your API keys
        nano backend/.env
        
        Update these lines:
        ALPACA_API_KEY=pk_live_YOUR_KEY_HERE
        ALPACA_SECRET_KEY=YOUR_SECRET_HERE
        STRIPE_SECRET_KEY=sk_live_YOUR_KEY_HERE
        PERSONA_API_KEY=YOUR_KEY_HERE
        ETHEREUM_RPC_URL=YOUR_RPC_URL_HERE
        
        Save: Ctrl+O → Enter → Ctrl+X

STEP 5: Deploy everything (one command!)
        chmod +x DEPLOY_NOW.sh
        ./DEPLOY_NOW.sh

STEP 6: Verify
        docker compose -f docker-compose.prod.yml ps
        curl https://xcapital.investments
